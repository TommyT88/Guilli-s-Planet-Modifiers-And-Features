### On colonized, set planet colour again if needed
# needed because when AI colonizes they rename the planet, so it needs to be recoloured and receive a new icon
# dont delete this again idiot ;)
planet_event = { id = gpm_colony.1 hide_window = yes is_triggered_only = yes
	trigger = {
		AND = {
			OR = {
				has_planet_flag = gpm_is_precursor_world
				has_planet_flag = gpm_is_wondrous_planet
			}
			owner = {
				is_ai = yes
				# new 21/04/2018 -guilli
				NOT = {
                    OR = {
                        is_country_type = swarm
                        is_country_type = extradimensional
                        is_country_type = extradimensional_2
                        is_country_type = extradimensional_3
                        is_country_type = ai_empire
                    }
                }
			}
		}
	}


	immediate = {
		# log = "---- Fired Event gpm_colony.1 ----"
		# log = "---- On Planet: [This.GetName] ----"
		# log = "---- set planet color on colonization(AI) ----"
	# gpm_reset_planet_name = yes
	# gpm_colour_planet_name = yes
		if = {
			limit = {
				has_planet_flag = gpm_is_precursor_world
			}
			set_name = {
				key = "NAME_color_precursor"
				variable_string = "[This.GetName]"
			}
		}
		if = {
			limit = {
				has_planet_flag = gpm_is_wondrous_planet
			}
			set_name = {
				key = "NAME_color_precursor"
				variable_string = "[This.GetName]"
			}
		}
	}
}

planet_event = {
	id = gpm_colony.3
	title = "gpm_colony.3.name"
	desc = {
		trigger = {
			owner = { NOT = { has_authority = auth_machine_intelligence	} }
		}
		text = gpm_colony.3.desc
	}
	desc = {
		trigger = {	owner = { has_authority = auth_machine_intelligence } }
		text = gpm_colony.3.desc.mach
	}
	picture = GFX_evt_gpm_spaceship_graveyard
	show_sound = event_radio_chatter_02
	location = root

	is_triggered_only = yes

	pre_triggers = {
		has_owner = yes
	}


	trigger = {
		num_pops > 0
		has_modifier = "gpm_spaceship_graveyard"
		NOT = {
			owner = { has_country_flag = gpm_colony_country_ship_event }
		}
	}


	immediate = {
		# log = "---- Fired Event gpm_colony.3 ----"
		# log = "---- On Planet: [This.GetName] ----"
		# log = "---- Ship graveyard event start ----"
		owner = { set_country_flag = gpm_colony_country_ship_event }
		begin_event_chain = {
			event_chain = "gpm_colony_ship_event_chain"
			target = root
		}
	}

	option = {
		trigger = {
			owner = { NOT = { has_authority = auth_machine_intelligence	} }
		}
		name = gpm_colony.3.a
		# planet_event = { id = gpm_colony.4 days = 30 }
		enable_special_project = {
			name = "GPM_COLONY_SHIP_3"
			location = this
			owner = root
		}
	}
	option = {
		trigger = {	owner = { has_authority = auth_machine_intelligence } }
		name = gpm_colony.3.mach
		# planet_event = { id = gpm_colony.4 days = 30 }
		enable_special_project = {
			name = "GPM_COLONY_SHIP_3"
			location = this
			owner = root
		}
	}
}

planet_event = {
	id = gpm_colony.7100
	hide_window = yes
	location = ROOT

	is_triggered_only = yes

	trigger = {
		exists = owner
		gpm_is_planet_fallen_empire = yes
		NOT = { has_planet_flag = gpm_developed_fallen_empire_world } # This excludes worlds the fallen empire already developed (incase of re-conquests)
		NOT = { has_global_flag = gpm_modifiers_fallen_empire_disabled }
	}

	immediate = {
		# log = "---- Fired Event gpm_colony.7100 ----"

		gpm_remove_negative_modifiers = yes
		gpm_remove_neutral_modifiers = yes
		# Spawn one guaranteed fallen empire modifier for colonies
		# Planet Modifier Scaling
		if = {
			limit = { NOT = { has_global_flag = gpm_scaling_multiplier_disabled } }
			gpm_scale_planet_modifiers = yes
		}
		gpm_roll_fallen_empire_world_modifiers = yes
		set_planet_flag = gpm_developed_fallen_empire_world
	}
}

# Gives special modifiers to non-machine worlds for machine empires.
planet_event = {
	id = gpm_colony.8000
	hide_window = yes
	location = ROOT

	is_triggered_only = yes

	trigger = {
		exists = owner
		gpm_is_planet_class_machine = no
		num_modifiers < 4
		OR = {
			owner = { has_authority = auth_machine_intelligence }
			owner = { has_authority = auth_ancient_machine_intelligence }
		}
	}

	immediate = {
		# log = "---- Fired Event gpm_colony.8000 ----"
		if = {
			limit = { owner = { has_authority = auth_ancient_machine_intelligence } }
			random_list = {
				0 = { gpm_roll_machine_world_modifiers = yes }
				90 = { }
			}
		}
		else_if = {
			limit = { owner = { has_authority = auth_machine_intelligence } }
			random_list = {
				10 = { gpm_roll_machine_world_modifiers = yes }
				90 = { }
			}
		}
	}
}
