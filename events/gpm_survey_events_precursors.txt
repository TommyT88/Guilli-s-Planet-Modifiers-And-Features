#1 spawn precursor dig site
ship_event = {
	id = gpm_survey_precursors.1
	hide_window = yes
	is_triggered_only = yes
	location = FROM

	trigger = {
		FROM = {
			gpm_has_precursor_modifiers = yes
			NOT = { exists = archaeological_site }
			NOT = { has_planet_flag = gpm_already_rolled_for_prec_dig_site }
			NOT = { has_planet_flag = gpm_machine_system_moon }
			NOT = { has_planet_flag = gpm_machine_system_machine_world }
			NOR = {
				gpm_is_planet_class_star = yes
				gpm_is_planet_class_asteroid = yes
				gpm_is_planet_class_gas_giant = yes
			}
		}
	}

	immediate = {
		locked_random_list = {
			30 = {
				FROM = { set_planet_flag = gpm_already_rolled_for_prec_dig_site }
				# modifier = { factor = 0 root.owner = { is_ai = yes } }
			}
			70 = {
				FROM = {
					create_archaeological_site = gpm_precursor_generic_digsite
					set_planet_flag = gpm_already_rolled_for_prec_dig_site
				}
				# set by the "extra fun" game option. Mostly for multiplayer by request
				modifier = { factor = 100 has_global_flag = gpm_multiplayer_fun_enabled }
			}
		}
	}
}

#10 gpm_precursor_remnants discovery
ship_event = {
	id = gpm_survey_precursors.10
	title = "gpm_survey_precursors.10.name"
	desc = {
		trigger = {
			owner = {
				NOR = {
					has_authority = auth_hive_mind
					has_authority = auth_machine_intelligence
				}
			}
		}
		text = "gpm_survey_precursors.10.desc"
	}
	desc = {
		trigger = {	owner = { has_authority = auth_hive_mind } }
		text = "gpm_survey_precursors.10.hive.desc"
	}
	desc = {
		trigger = {	owner = { has_authority = auth_machine_intelligence } }
		text = "gpm_survey_precursors.10.mach.desc"
	}
	picture = GFX_evt_gpm_precursor_remnants
	show_sound = event_scanner
	location = FROM

	is_triggered_only = yes

	trigger = {
		FROM = {
			AND = {
				has_modifier = "gpm_precursor_remnants"
				root = { owner = { NOT = { has_country_flag = gpm_discover_precursor_remnants } } }
			}
		}
	}

	immediate = {
		root = { owner = { set_country_flag = gpm_discover_precursor_remnants } }
		FROM = {
			if = {
				limit = { NOT = { has_planet_flag = "gpm_is_surveyed" }	}
				gpm_colour_planet_name = yes
				set_planet_flag = "gpm_is_surveyed"
			}
			save_event_target_as = precursor_planet
		}
		root.leader = {
			locked_random_list = {
				5 = {
					add_trait = leader_trait_gpm_precursor_expert
					hidden_effect = { owner = { set_country_flag = gpm_precursor_expert_trait_rewarded } }
					modifier = { factor = 0 has_trait = leader_trait_gpm_precursor_expert }
				}
				95 = { }
			}
		}
	}

	# Base Reward Option
	option = {
		# trigger = {
		# 	root.owner = { NOT = { has_authority = 000000000000000000 } }
		# }

		name = FASCINATING
		# leader experience
		# random big experience, double if exploraction policy = slow explore
		custom_tooltip = gpm_survey_reward_leader_xp
		root.leader = { gpm_random_experience_big = yes }

		# resources
		custom_tooltip = gpm_survey_reward_empire
		root.owner = {
			add_monthly_resource_mult = {
				resource = unity
				value = @tier1unityreward_half
				min = @tier1unitymin_half
				max = @tier1unitymax_half
			}
		}

		# random reward from reward list
		root.owner = {
			gpm_random_reward_precursor_remnants = yes
		}

		# aditional reward from exploration policy
		root.owner = {
			if = {
				limit = { has_policy_flag = gpm_more_discovery_rewards }
				custom_tooltip = gpm_survey_reward_bonus
				gpm_exploration_policy_extra_reward = yes
			}
		}
	}
	# Megacorp Reward Option
	option = {
		trigger = {
			root.owner = { has_authority = auth_corporate }
		}

		name = gpm_survey.MEGACORPSELL
		# leader experience
		# random big experience, double if exploraction policy = slow explore
		custom_tooltip = gpm_survey_reward_leader_xp
		root.leader = { gpm_random_experience_big = yes }

		# planet deposit:
		FROM = {
			if = {
				limit = { NOT = { has_deposit = d_trade_value_6 } }
				custom_tooltip = gpm_survey_reward_deposit
				add_deposit = d_trade_value_6
			}
		}
		# resources
		custom_tooltip = gpm_survey_reward_empire
		custom_tooltip = gpm_survey_reward_energy_megacorp
		root.owner = {
			locked_random_list = {
				11 = { add_resource = { energy = 200 } }
				22 = { add_resource = { energy = 300 } }
				33 = { add_resource = { energy = 350 } }
				44 = { add_resource = { energy = 400 } }
				33 = { add_resource = { energy = 450 } }
				22 = { add_resource = { energy = 500 } }
				11 = { add_resource = { energy = 600 } }
			}
		}

		# random reward from reward list
		root.owner = {
			gpm_random_reward_precursor_remnants = yes
		}

		# aditional reward from exploration policy
		root.owner = {
			if = {
				limit = { has_policy_flag = gpm_more_discovery_rewards }
