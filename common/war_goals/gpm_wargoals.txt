# r_dragon_trophy
wg_gpm_r_dragon_trophy = {
	casus_belli = cb_gpm_r_dragon_trophy

	hide_if_no_cb = yes
	release_occupied_systems_on_status_quo = no

	galactic_empire_joins_defender = {
		from = { is_galactic_emperor = yes }
		imperial_authority >= 100
	}
	
	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}

	surrender_acceptance = -10


	potential = {
		always = yes
	}
	possible = {
		custom_tooltip = {
			fail_text = has_claimed_all_systems
			from = {
				is_country_type = default
				any_owned_planet = {
					is_colony = yes
					root = {
						NOT = { has_claim = prev.solar_system }
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = cannot_use_against_emperor
			OR = {
				from = { is_galactic_emperor = no }
				is_galactic_community_member = no
			}
		}
		custom_tooltip = {
			fail_text = pax_galactica_active
			NAND = {
				is_active_resolution = resolution_emperor_pax_galactica
				is_galactic_community_member = yes
				from = { is_galactic_community_member = yes }
			}
		}
	}
	on_accept = {
		hidden_effect = {
			set_timed_country_flag = {
				flag = gpm_recent_relic_war_ended
				years = 20
			}
			from = {
				add_threat = {
					who = root
					amount = 0.5
				}
			}
		}
		# Relic switches hands
		if = {
			limit = { from = { has_relic = r_dragon_trophy } }
			from = { remove_relic = r_dragon_trophy }
			add_relic = r_dragon_trophy
			log = ";RELIC WAR; [from.GetName] lost his r_dragon_trophy relic to empire [root.GetName] "
			# flag operations
			from = { remove_country_flag = gpm_has_r_dragon_trophy_relic@this }  # removing to clean up
			set_country_flag = gpm_has_r_dragon_trophy_relic@from  # losing empire gets this flag because it knows we have this relic
		}
		# Defender is humiliated
		from = { add_modifier = { modifier = humiliated days = 3600 } }
		log = ";RELIC WAR; [from.GetName] humiliated "
	}	
	on_status_quo = {
		log = ";RELIC WAR; defender [from.GetName] status quo'd with attacker [root.GetName]"
		from = {
			set_closed_borders = {
				who = prev
				status = no
			}
		}
	}

	ai_weight = {
		weight = 10
		modifier = {
			factor = 0
			has_country_flag = gpm_recent_relic_war_ended
		}
		modifier = {
			factor = 2
			is_rival = from
		}
		modifier = {
			opinion_level = {
				who = from
				level <= neutral
			}
			factor = 1.5
		}
		modifier = {
			owner = { num_owned_relics = 1 }
			factor = 1.2
		}
		modifier = {
			owner = { num_owned_relics = 2 }
			factor = 1.4
		}
		modifier = {
			owner = { num_owned_relics = 3 }
			factor = 1.6
		}
		modifier = {
			owner = { num_owned_relics = 4 }
			factor = 1.8
		}
		modifier = {
			owner = { num_owned_relics = 5 }
			factor = 2
		}
		modifier = {
			owner = { num_owned_relics >= 6 }
			factor = 2.2
		}
		modifier = {
			factor = 2
			has_ethic = ethic_militarist
		}
		modifier = {
			factor = 4
			has_ethic = ethic_fanatic_militarist
		}
		modifier = {
			factor = 2
			has_ethic = ethic_materialist
		}
		modifier = {
			factor = 4
			has_ethic = ethic_fanatic_materialist
		}
		# modifier = {
		# 	factor = 0
		# 	opinion_level = {
		# 		who = root
		# 		level >= good
		# 	}
		# }
		# modifier = {
