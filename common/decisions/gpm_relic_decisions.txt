	sound = event_administrative_work
	icon = gpm_decision_machine_world_core_relic

    potential = {
		exists = space_owner
		space_owner = { has_country_flag = relic_machine_world_core_activated }

			gpm_is_planet_class_normal_uninhabitable = yes
			gpm_is_planet_class_gas_giant = yes
			is_planet_class = pc_broken
		}
	}
	
	allow = {
		custom_tooltip = decision_gpm_machine_world_core_relic_tt
	}

    effect = {
		hidden_effect = { space_owner = { remove_country_flag = relic_machine_world_core_activated } }
		custom_tooltip = decision_gpm_machine_world_core_relic_tt
		change_pc = pc_machine
		hidden_effect = {
			reset_planet = yes
			clear_deposits = yes
			# switch "fixable" negative modifiers with their fixed variant
			gpm_switch_fixable_modifiers = yes
			# Remove bad modifiers where is makes sense
			gpm_remove_negative_modifiers = yes
			# Remove neutral modifiers where it makes sense
			gpm_remove_neutral_modifiers = yes
			# Remove good modifiers where it makes sense
			gpm_remove_positive_modifiers = yes
		}
		change_planet_size = 5
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 1000
			AND = {
				planet_size >= 20
				gpm_is_planet_class_normal_uninhabitable = yes
			}
		}
	}
}
