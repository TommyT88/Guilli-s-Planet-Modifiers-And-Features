gpm_roll_star_modifiers = {
	optimize_memory
	switch = {
		trigger = has_global_flag
		gpm_modifier_amount_max_one = {
			# spawn random modifiers and ignore PD clasification modifiers in the counting
			if = {
				limit = { gpm_has_pd_planet_modifier = yes }
				while = {
					count = 2
					if = {
						limit = { num_modifiers < 2 }
						random_list = {
							5 = { }
							95 = {
								gpm_spawn_star_modifier = yes
								modifier = { factor = 0.25 num_modifiers = 2 }
								modifier = { factor = 0.10 num_modifiers = 3 }
							}
						}
					}
				}
			}
			if = {
				limit = { gpm_has_pd_planet_modifier = no }
				while = {
					count = 1
					if = {
						limit = { num_modifiers < 1 }

							10 = { }
							90 = {
								gpm_spawn_star_modifier = yes
