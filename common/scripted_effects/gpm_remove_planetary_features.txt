gpm_remove_random_vanilla_planetary_feature = {
    random_deposit = {
        limit = {
            is_blocker = no
            gpm_is_vanilla_feature = yes
            PREV = { NOT = { has_planet_flag = gpm_is_terraformed_world } }
        }
        remove_deposit = yes
    }
}
