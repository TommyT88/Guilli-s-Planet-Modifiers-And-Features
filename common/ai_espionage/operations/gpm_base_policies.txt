gpm_spy_policy_steal_relic = {
	weight = {
		base = 400
		### DEBUGGING ###
		### END DEBUGGING ###
		# Dont let AI use this on their subjects or their overlord
		modifier = {
			owner = { is_overlord_to = root.target	}
			factor = 0
		}
		modifier = {
			root.target = { is_overlord_to = owner	}
			factor = 0
		}
		# Weigts bases on oppinion levels. We dont steal from friends.
		modifier = {
			owner = {
				OR = {
					is_friendly_to = root.target
					is_loyal_to = root.target
					is_protective_to = root.target
					is_patronizing_to = root.target
					is_custodial_to = root.target
				}
			}
			factor = 0
		}
		
		modifier = {
			owner = {
				OR = {
					is_neutral_to = root.target
					is_dismissive_to = root.target
					is_cordial_to = root.target
					is_enigmatic_to = root.target
					is_arrogant_to = root.target
				}
			}
			factor = 0.5
		}
		modifier = {
			owner = {
				OR = {
					is_unfriendly_to = root.target
					is_disloyal_to = root.target
					is_threatened_to = root.target
					is_angry_to = root.target
					is_domineering_to = root.target
					is_imperious_to = root.target	
				}
			}
			factor = 1.5
		}
		modifier = {
			owner = {
				OR = {
					is_hostile_to = root.target
					is_belligerent_to = root.target
					is_berserker_to = root.target
					is_disloyal_to = root.target
				}
			}
			factor = 2
		}
		# Weights based on other empires fleet power. We dont steal if we can end up in a fight we cant possibly win
		modifier = {
			owner = {
				relative_power = {
					who = root.target
					category = fleet
					value = overwhelming
				}
			}
			factor = 0.1
		}
		modifier = {
			owner = {
				relative_power = {
					who = root.target
					category = fleet
					value = superior
				}
			}
			factor = 0.5
		}
		modifier = {
			owner = {
				relative_power = {
					who = root.target
					category = fleet
					value = inferior
				}
			}
			factor = 1.1
		}
		modifier = {
			owner = {
				relative_power = {
					who = root.target
					category = fleet
					value = pathetic
				}
			}
			factor = 1.2
		}
		# Added weight if rival because fuck you empire X
		modifier = {
			owner = {
				has_rival = root.target
			}
			factor = 2
		}
		# With every relic our taste for more increases
		modifier = {
			owner = { num_owned_relics = 0 }
			factor = 0
		}
		modifier = {
			owner = { num_owned_relics = 1 }
			factor = 1.1
		}
		modifier = {
			owner = { num_owned_relics = 2 }
			factor = 1.2
		}
		modifier = {
			owner = { num_owned_relics = 3 }
			factor = 1.3
		}
		modifier = {
			owner = { num_owned_relics = 4 }
			factor = 1.4
		}
		modifier = {
			owner = { num_owned_relics >= 5 }
			factor = 1.5
		}
		# some empires like relics a little more than others
		modifier = {
			factor = 1.25
			owner = { has_ethic = ethic_militarist }
		}
		modifier = {
			factor = 1.5
			owner = { has_ethic = ethic_fanatic_militarist }
		}
		modifier = {
			factor = 1.5
			owner = { has_ethic = ethic_materialist }
		}
		modifier = {
			factor = 2
			owner = { has_ethic = ethic_fanatic_materialist }
		}
		# Disable if the target or the targetting empire has "recently" attempted or been victim of a steal relic operation
		modifier = {
			factor = 0
			OR = {
				owner = { has_country_flag = gpm_recent_heist }
				root.target = { has_country_flag = gpm_recent_heist }
			}
		}
		# we help the player a little, it's more fun to see AI's screw eachother over than to be screwed over
		modifier = {
			factor = 0.1
			root.target = { is_ai = no }
		}
		# added weights based on difficulty level
		# Cadet
		modifier = {
			is_difficulty = 0
			factor = 0.50
		}
		# Ensign
		modifier = {
			is_difficulty = 1
			factor = 0.75
		}
		# Captain
		modifier = {
			is_difficulty = 2
			factor = 1
		}
		# Commodore
		modifier = {
			is_difficulty = 3
			factor = 1.25
		}
		# Admiral
		modifier = {
			is_difficulty = 4
			factor = 1.5
		}
		# Grand Admiral
		modifier = {
			is_difficulty = 5
			factor = 2
		}
		# Disable if the game option is enabled
		modifier = {
			factor = 0
			AND = {
				root.target = { is_ai = no }
				has_global_flag = gpm_relic_heists_AI_disabled
			}
		}
		# diplomatic modfiers
		modifier = {
			factor = 0
			owner = {
				OR = {
					has_non_aggression_pact = root.target
					has_defensive_pact = root.target
				}
			}
		}
		modifier = {
			factor = 0
			owner = { is_improving_relations_with = root.target }
		}
		modifier = {
			factor = 0
			root.target = { is_loyal_to = owner }
		}
	}
}
